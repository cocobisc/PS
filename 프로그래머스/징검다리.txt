#include <string>
#include <vector>
#include <algorithm>
#include <iostream>
using namespace std;

int solution(int distance, vector<int> t, int n) {
    t.push_back(distance);
    vector<int> arr(1);
    for(auto i:t) arr.push_back(i);
    sort(arr.begin(), arr.end());

    int le = 0, ri = 1e9 + 2;

    while(le <= ri) {
        int mid = le + ri >> 1;
        int k, cnt = 0;
        int sidx = 0;
        for(int i=1;i<arr.size();i++) {
            k = arr[i] - arr[sidx];
            if(k < mid) cnt++;
            else sidx = i;
        }
        if(cnt > n) ri = mid - 1;
        else le = mid + 1;
    }
    return ri;
}
