#include <cstdio>
#include <cstdlib>

int n;
int* ft_range(int min, int max) {
	if (min >= max) return NULL;

	scanf("%d", &n);
	int *arr = (int*)malloc(sizeof(int) * n);
	for (int i = 0; i < n; i++) scanf("%d", &arr[i]);

	int *range_arr = (int*)malloc(sizeof(int) * (n + 1));
	int idx = 1;
	for (int i = 0; i < n; i++) {
		if (arr[i] >= min && arr[i] < max) range_arr[idx++] = arr[i];
	}
	range_arr[0] = idx;
	return range_arr;
}
int main() {
	int *arr = ft_range(5, 5);
	if (arr == NULL) return 0;

	for (int i = 1; i < arr[0]; i++)
		printf("%d ", arr[i]);
	return 0;
}
