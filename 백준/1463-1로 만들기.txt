#include <iostream>
#include <algorithm>
using namespace std;
int n, d[1000001], m=999999999;
int main() {
	cin >> n;
	d[1] = 0;
	d[2] = d[3] = 1;
	for (int i = 4; i <= n; i++) d[i] = i;
	for (int i = 4; i <= n; i++) {
		if (i % 3 == 0 && i % 2 == 0) d[i] = min(d[(int)i / 3], d[(int)i / 2]);
		else if (i % 2 == 0) d[i] = d[(int)i / 2];
		else if (i % 3 == 0) d[i] = d[(int)i / 3];
		d[i] = min(d[i], d[i - 1]);
		d[i]++;
	}
	cout << d[n] << endl;
}
