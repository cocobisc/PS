#include <bits/stdc++.h>
using namespace std;
const int MAX = 3e5 + 1;

int N, K, L, R;
int a[MAX], alen[21];
long long ans;
string s;

int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL);

	cin >> N >> K;
	while (R < N) {
		cin >> s;
		a[R] = s.size();
		ans += alen[a[R]]++;
		if (++R - L > K) alen[a[L++]]--;
	}
	cout << ans;

	return 0;
}
