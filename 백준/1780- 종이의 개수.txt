#include <bits/stdc++.h>
using namespace std;

const int MAX = 2200;
int a[3], arr[MAX][MAX];

bool check(int x1, int y1, int x2, int y2) {
	int k = arr[y1][x1];
	for (int i = y1; i < y2; i++) {
		for (int j = x1; j < x2; j++) if (arr[i][j] != k) return 0;
	}
	a[k + 1]++;
	return 1;
}

void findRC(int x1, int y1, int x2, int y2) {
	int k = (x2 - x1) / 3;
	if (!check(x1, y1, x2, y2)) {
		for (int i = 0; i < 3; i++) {
			for (int j = 0; j < 3; j++) {
				findRC(x1 + j * k, y1 + i * k, x1 + (j + 1) * k, y1 + (i + 1) * k);
			}
		}
	}
}

int main() {
	int n; cin >> n;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) cin >> arr[i][j];
	}
	findRC(0, 0, n, n);
	for (int i = 0; i < 3; i++) {
		cout << a[i] << '\n';
	}
}
