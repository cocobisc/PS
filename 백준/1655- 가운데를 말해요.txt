#include <bits/stdc++.h>
using namespace std;
int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	priority_queue<int, vector<int>, greater<int>> pq2;
	priority_queue<int> pq1;
	int n; cin >> n;
	for (int i = 0; i < n; i++) {
		int a; cin >> a;
		if (pq2.empty()) pq2.push(a);
		else {
			if (pq2.size() == pq1.size()) {
				if (pq1.top() > a) {
					pq2.push(pq1.top());
					pq1.pop(); pq1.push(a);
				}
				else pq2.push(a);
			}
			else {
				if (pq2.top() > a) pq1.push(a);
				else {
					pq2.push(a);
					pq1.push(pq2.top());
					pq2.pop();
				}
			}
		}
		if (pq2.size() == pq1.size()) 
			cout << min(pq1.top(), pq2.top());
		else cout << pq2.top();
		cout << "\n";
	}
}
