#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int n, arr[1001], dp[1001];
int main(void) {
	cin >> n;
	for (int i = 1; i <= n; i++) cin >> arr[i];
	for (int i = 1; i <= n; i++) {
		for (int j = i; j <= n; j++) {
			dp[j] = max(dp[j], dp[j - i] + arr[i]);
		}
	}
	cout << dp[n] << endl;
	return 0;
}
