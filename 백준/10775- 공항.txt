#include <bits/stdc++.h>
using namespace std;
const int MAX = 1e5;
int parent[MAX];

int find(int x) {
	if (parent[x] == -1) return x;
	return parent[x] = find(parent[x]);
}

void uni(int a, int b) {
	parent[b] = a;
}

int main() {
	int G, P; cin >> G >> P;
	int arr[MAX];
	memset(parent, -1, sizeof(parent));
	int ans = 0;
	for (int i = 0; i < P; i++) cin >> arr[i];
	for (int i = 0; i < P; i++) {
		int b = find(arr[i]);
		int a = find(arr[i] - 1);
		if (a != b) uni(a, b);
		else {
			if (!a && !b) break;;
			uni(a - 1, a);
		}
		ans++;
	}
	cout << ans;
}
