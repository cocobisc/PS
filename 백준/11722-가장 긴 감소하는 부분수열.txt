#include <iostream>
#include <algorithm>
#include <vector>
#include <algorithm>
using namespace std;
int n, arr[1000];
int main(void) {
	cin >> n;
	for (int i = 0; i < n; i++) cin >> arr[i];
	vector<int> v;
	v.push_back(arr[0]);
	for (int i = 1; i < n; i++) {
		for (int j = 0; j < v.size(); j++) {
			if (v[j] <= arr[i]) {
				v[j] = arr[i];
				break;
			}
			if (j == v.size() - 1 && v[j] > arr[i]) v.push_back(arr[i]);
		}
	}
	cout << v.size() << endl;
	
	return 0;
}
