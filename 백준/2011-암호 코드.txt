#include <iostream>
#include <string>
using namespace std;
int dp[5001];

int main() {
	string s;
	cin >> s;
	dp[0] = dp[1] = 1;
	for (int i = 2; i <= s.size(); i++) {
		if (s[0] == '0') break;
		if (s[i - 1] != '0') dp[i] = dp[i - 1] % 1000000;
		int t = (s[i - 2] - '0') * 10 + s[i - 1] - '0';
		if (10 <= t && t <= 26) dp[i] = (dp[i] + dp[i - 2]) % 1000000;
	}
	if(s[0] != '0') cout << dp[s.size()] << endl;
	else cout << 0 << endl;
}
