#include <iostream>
using namespace std;
int n, dp[1001][10];
int main() {
	cin >> n;
	for (int i = 0; i < 10; i++) dp[1][i] = 1;
	for (int i = 2; i <= n; i++) {
		int sum = 0;
		for (int j = 0; j < 10; j++) {
			sum += dp[i - 1][j];
			dp[i][j] += sum;
			dp[i][j] %= 10007;
		}
	}
	int ans = 0;
	for (int i = 0; i < 10; i++) ans += dp[n][i];
	cout << ans % 10007 << endl;
}
