#include <bits/stdc++.h>
using namespace std;
const int MAX = 1e6;
int fail[MAX];

int main() {
	string a, b;
	getline(cin, a);
	getline(cin, b);
	
	for (int i = 1, j = 0; i < b.size(); i++) {
		while (j > 0 && b[i] != b[j]) j = fail[j - 1];
		if(b[i] == b[j]) fail[i] = ++j;
	}

	vector<int> ans;
	for (int i = 0, j = 0; i < a.size(); i++) {
		while (j > 0 && a[i] != b[j]) j = fail[j - 1];
		if (a[i] == b[j]) {
			if (j == b.size() - 1) {
				ans.push_back(i - b.size() + 2);
				j = fail[j];
			}
			else j++;
		}
	}
	cout << ans.size() << '\n';
	for (auto i : ans) cout << i << '\n';
}
