#include <iostream>
using namespace std;
int n;
long long dp[101][10], temp[10];
int main() {
	cin >> n;
	for (int i = 1; i < 10; i++) dp[1][i] = 1;
	for (int i = 2; i <= n; i++) {
		for (int j = 0; j < 10; j++) {
			if (j == 9) temp[j] = dp[i - 1][8];
			else if (j == 0) temp[j] = dp[i - 1][1];
			else temp[j] = dp[i - 1][j - 1] + dp[i - 1][j + 1];
		}
		for (int j = 0; j < 10; j++) dp[i][j] = temp[j] % 1000000000;
	}
	int sum = 0;
	for (int i = 0; i < 10; i++) {
		sum += dp[n][i];
		sum %= 1000000000;
	}
	cout << sum << endl;
}
