#include <bits/stdc++.h>
using namespace std;

int main() {
	int N, K; cin >> N >> K;
	vector<pair<int, int>> jew;
	vector<int> bp;
	for (int i = 0; i < N; i++) {
		int a, b; cin >> a >> b;
		jew.push_back({ a,b });
	}
	for (int i = 0; i < K; i++) {
		int a; cin >> a;
		bp.push_back(a);
	}
	sort(jew.begin(), jew.end());
	sort(bp.begin(), bp.end());

	priority_queue<int> pq;
	int idx = 0;
	long long ans = 0;
	for (int i = 0; i < K; i++) {
		while (idx < N && jew[idx].first <= bp[i]) 
			pq.push(jew[idx++].second);
		if(pq.size()) ans += pq.top(), pq.pop();
	}
	cout << ans;

	return 0;
}
