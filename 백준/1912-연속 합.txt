#include <iostream>
#include <algorithm>
#include <vector>
#include <algorithm>
using namespace std;
int n, arr[100000], dp[100000];
int main(void) {
	cin >> n;
	for (int i = 0; i < n; i++) cin >> arr[i];
	dp[0] = arr[0];
	int ret = 0;
	for (int i = 1; i < n; i++) {
		dp[i] = max(dp[i - 1] + arr[i], arr[i]);
		if (ret < dp[i]) ret = dp[i];
	}
	cout << ret << endl;

	return 0;
}
