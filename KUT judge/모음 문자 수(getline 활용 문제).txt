#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
string t;
char a[5] = { 'a', 'e', 'i', 'o', 'u' };
int main() {
	getline(cin, t);
	int n = atoi(t.c_str());
	for (; n--;) {
		string s;
		int cnt = 0;
		getline(cin, s);
		for (int i = 0; i < s.size(); i++) {
			for (int j = 0; j < 5; j++)
				if (s[i] == a[j]) cnt++;
		}
		cout << cnt << endl;
	}
}
