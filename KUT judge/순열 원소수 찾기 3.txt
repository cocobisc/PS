#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <math.h>
typedef unsigned long long ull;
using namespace std;
int main() {
	int num;
	cin >> num;
	for (int i = 0; i < num; i++) {
		int n;
		ull k;
		cin >> n >> k;
		ull s = pow(k, 1 / (float)n);
		for (;;s++) {
			long long r = s;
			if (n == 2) r *= r - 1;
			else if (n == 3) r *= (r - 1) * (r - 2);
			else r *= (r - 1) * (r - 2) * (r - 3);
			if (r == k) {
				cout << s - n << "\n";
				break;
			}
		}
	}
	return 0;
}
