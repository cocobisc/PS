#define _CRT_SECURE_NO_WARNINGS
#include <cstdio>
#include <algorithm>
#include <vector>
#include <cstring>

using namespace std;
typedef pair<int, int> pii;
int t;

int main() {
	scanf("%d", &t);
	while (t--) {
		vector<pii> v;
		int n, d, p;
		bool flag = false;
		scanf("%d %d %d", &n, &d, &p);

		for (int i = 0; i < n; i++) {
			int a;
			scanf("%d", &a);
			v.push_back(pii(a, i));
		}
		sort(v.begin(), v.end());
		for (int i = 0; i < n; i++) {
			for (int j = i + 1; j < n; j++) {
				int s = abs(v[i].first - v[j].first);
				if (s > p) break;
				if (abs(v[i].second - v[j].second) <= d) {
					flag = true;
					break;
				}
			}
			if (flag) break;
		}
		if (flag) printf("TRUE\n");
		else printf("FALSE\n");
	}
}
