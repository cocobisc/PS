#define	_CRT_SECURE_NO_WARNINGS
#include <iostream>
using namespace std;
typedef unsigned long long ull;
int tt;
ull d[101];
int main() {
	cin >> tt;
	while (tt--) {
		int n;
		cin >> n;
		d[1] = 1;
		d[2] = 3;
		for (int i = 3; i <= n; i++) {
			d[i] = d[i - 1] + d[i - 2] * 2;
			d[i] %= 1000000007;
		}
		cout << d[n] << endl;
	}
}
