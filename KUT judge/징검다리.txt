#define _CRT_SECURE_NO_WARNINGS
#include<cstdio>
#define min(a,b) (a<b?a:b)
int main() {
	int t; scanf("%d", &t);
	while (t--) {
		int n, arr[10001], dp[10001] = { 0, };
		scanf("%d\n", &n);
		for (int i = 1; i <= n; i++) scanf("%d", &arr[i]);
		dp[1] = arr[1];
		dp[2] = arr[2];
		for (int i = 3; i <= n; i++) dp[i] = min(dp[i - 1], min(dp[i - 2], dp[i - 3])) + arr[i];
		if (n < 2) printf("0\n");
		else printf("%d\n", min(dp[n - 2], min(dp[n - 1], dp[n])));
	}
}
