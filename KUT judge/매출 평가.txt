#include<cstdio>
int main() {
	int t; scanf("%d", &t);
	while (t--) {
		int n, arr[1000], ans = 0;
		scanf("%d", &n);
		for (int i = 0; i < n; i++) scanf("%d", &arr[i]);
		for (int i = 1; i < n; i++) {
			int cnt = 0;
			for (int j = i - 1; j >= 0; j--) {
				if (arr[i] >= arr[j]) cnt++;
			}
			ans += cnt;
		}
		printf("%d\n", ans);
	}
}
