#define _CRT_SECURE_NO_WARNINGS
#include <cstdio>

int main() {
	int t; scanf("%d", &t);

	while(t--) {
		int n;
		scanf("%d", &n);
		bool flag = false;
		for (int i = n; i >= 0; i--) {
			int sum = i, t = i;
			while (t > 0) { sum += t % 10; t /= 10; }
			if (sum < n - 20) break;
			if (sum == n) {
				flag = true;
				break;
			}
		}
		if (flag) printf("No\n");
		else printf("Self\n");
	}
}
