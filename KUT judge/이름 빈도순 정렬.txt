#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#define MAX 20001
using namespace std;
int alpha[26];
string s[MAX];

bool cmp(string a, string b) {
	if (alpha[a.at(0) - 'a'] != alpha[b.at(0) - 'a'])
		return alpha[a.at(0) - 'a'] > alpha[b.at(0) - 'a'];
	else
		return a < b;
}

int main() {
	int num;
	cin >> num;
	for (int i = 0; i < num; i++) {
		cin >> s[i];
		alpha[s[i].at(0) - 'a']++;
	}
	sort(s, s + num, cmp);
	for (int i = 0; i < num - 1; i++) cout << s[i] << "\n";
	cout << s[num - 1];
}
