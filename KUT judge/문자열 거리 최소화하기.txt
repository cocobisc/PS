#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <string>
using namespace std;
int n;
int main() {
	cin >> n;
	while (n--) {
		string a, b;
		cin >> a >> b;
		int min = 1001;
		for (int i = 0; i < b.size() - a.size() + 1; i++) {
			int cnt = 0;
			for (int j = 0; j < a.size(); j++) {
				if (a[j] != b[i + j]) cnt++;
			}
			if (cnt < min) min = cnt;
		}
		printf("%d\n", min);
	}
}
