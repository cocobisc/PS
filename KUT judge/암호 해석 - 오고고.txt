#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include <string>
using namespace std;
int main() {
	int i = 0, n; cin >> n;
	string s, ans; cin >> s;
	int t;
	while(i < s.size()) {
		t = i;
		if (s[i] == 'o') {
			while (i + 2 < s.size()) {
				if (s[i + 1] != 'g' || s[i + 2] != 'o') break;
				i += 2;
			}
		}
		if (t != i) for (int z = 3; z--;) ans.push_back('*');
		else ans.push_back(s[t]);
		i++;
	}
	cout << ans << endl;
}
