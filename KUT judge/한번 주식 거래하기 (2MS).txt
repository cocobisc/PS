#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <cstdio>
#include <string>
#include <vector>
#define MAX 501
#define MIN 1000001
#define swap(a,b) {char temp = a; a = b; b = temp;}
typedef unsigned long long ull;
using namespace std;
int n, i;
int main() {
	cin >> n;
	int min[10001] = {};
	int max[10001] = {};
	int arr[10001] = {};
	int mi = MIN, ma = 0;
	for (int i = 0; i < n; i++) {
		scanf("%d", &arr[i]);
		if (arr[i] < mi) {
			mi = arr[i];
			min[i] = arr[i];
		}
		else min[i] = mi;
	}
	for (int i = n - 1; i >= 0; i--) {
		if (arr[i] > ma) {
			ma = arr[i];
			max[i] = arr[i];
		}
		else max[i] = ma;
	}
	ma = 0;
	for (int i = 0; i < n - 1; i++) {
		if (max[i + 1] - min[i] > ma) ma = max[i + 1] - min[i];
	}
	
	return 0;
}
