#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#define swap(a,b) {int t = a; a = b; b = t;}
using namespace std;
bool isprime(int n) {
	for (int i = 2; i*i <= n; i++) if (!(n % i)) return false;
	return true;
}

void print(int n, int digit) {
	if (digit == 1) {
		printf("%d\n", n);
		return;
	}
	n *= 10;
	for (int i = 1; i <= 9; i++) {
		n += i;
		if (isprime(n)) print(n, digit - 1);
		n -= i;
	}
}

int main() {
	int digit;
	scanf("%d", &digit)

	print(2, digit);
	print(3, digit);
	print(5, digit);
	print(7, digit);

	return 0;
}
