#include <bits/stdc++.h>
using namespace std;
int dp[100001][4];
int main() {
	int a, n;
	scanf("%d %d", &n, &a);
	dp[1][0] = -a;
	for (int i = 2; i <= n; i++) {
		scanf("%d", &a);
		dp[i][0] = max(dp[i - 1][0], -a);
		dp[i][1] = max(a + dp[i - 1][0], dp[i - 1][1]);
		dp[i][2] = max(dp[i - 1][1] - a, dp[i - 1][2]);
		if(dp[i][2]) dp[i][3] = max(a + dp[i - 1][2], dp[i - 1][3]);
	}
	printf("%d", max(dp[n][1], dp[n][3]));
}
