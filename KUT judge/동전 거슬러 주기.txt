#define	_CRT_SECURE_NO_WARNINGS
#include <iostream>
using namespace std;
typedef unsigned long long ull;
int tt;

int main() {
	cin >> tt;
	while (tt--) {
		int n, coin[1000], change, dp[10001] = { 0, };
		cin >> n;
		for (int i = 0; i < n; i++) cin >> coin[i];
		cin >> change;
		dp[0] = 1;
		for (int i = 0; i < n; i++) {
			for (int j = coin[i]; j <= change; j++) {
				dp[j] += dp[j - coin[i]];
				dp[j] %= 1000000007;
			}
		}
		cout << dp[change] << endl;
	}
}
