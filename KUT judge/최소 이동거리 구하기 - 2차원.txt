#include <iostream>
#include <algorithm>
using namespace std;
int n, x[100000], y[100000];
int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		int a, b; cin >> a >> b;
		x[i] = a; y[i] = b;
	}
	sort(x, x + n); sort(y, y + n);
	int m1 = x[n / 2], m2 = y[n / 2], a = 0;
	for (int i = 0; i < n; i++) 
		a += abs(m1 - x[i]) + abs(m2 - y[i]);
	cout << a;
}
