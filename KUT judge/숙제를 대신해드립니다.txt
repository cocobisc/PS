#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <queue>
#include <cstdio>
#include <string>
#include <vector>
#define MAX 501
#define MIN 1000001
#define swap(a,b) {char temp = a; a = b; b = temp;}
typedef unsigned long long ull;
using namespace std;
int n, k;
vector<pair<int, int>> v;
priority_queue<int, vector<int>, greater<int>> pq;
int bot[11] = {};
int main() {
	cin >> n >> k;
	for (int i = 0; i < k; i++) {
		int a, b;
		cin >> a >> b;
		v.push_back(pair<int, int>(a, b));
	}
	int max = 0;
	for (int i = 0; i < k; i++) {
		if (bot[i % n] <= v[i].first) bot[i % n] = v[i].first + v[i].second;
		else bot[i % n] += v[i].second;
		if (max < bot[i % n]) max = bot[i % n];

		if (pq.empty()) pq.push(v[i].first + v[i].second);
		else {
			if (pq.top() > v[i].first) {
				if (pq.size() == n) {
					if (pq.top() <= v[i].first) {
						pq.push(v[i].first + v[i].second);
						pq.pop();
					}
					else {
						int temp = pq.top() + v[i].second;
						pq.pop();
						pq.push(temp);
					}
				}
				else pq.push(v[i].first + v[i].second);	
			}
			else {
				pq.push(v[i].first + v[i].second);
				pq.pop();
			}
		}
	}
	int max2;
	while (!pq.empty()) {
		max2 = pq.top();
		pq.pop();
	}
	cout << (max - max2) * 100 << endl;

	return 0;
}
