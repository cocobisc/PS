#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <string>
using namespace std;
int n;
int main() {
	cin >> n;
	for (; n--;) {
		string a, b;
		cin >> a >> b;
		int c1 = 0, c2 = 0;
		for (int i = 0; i < b.size(); i++) {
			if (b[i] == 'L') c1++;
			else c2++;
		}
		int idx = c1 - c2;
		if (idx < 0) idx = a.size() + idx;
		for (int i = 0; i < a.size(); i++) printf("%c", a[(idx + i) % a.size()]);
		printf("\n");
	}
}
