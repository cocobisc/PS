#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include <string>
using namespace std;
int arr[1000][1000];
int main() {
	int r, c;
	cin >> r >> c;
	for (int i = 0; i < r; i++) {
		for (int j = 0; j < c; j++) cin >> arr[i][j];
	}
	
	int cnt = 0;
	for (int i = 0; i < r; i++) {
		for (int j = 0; j < c; j++) {
			if (!arr[i][j]) {
				for (int k = 0; k < i; k++) {
					if (arr[k][j]) {
						cnt++; break;
					}
				}
				for (int k = i + 1; k < r; k++) {
					if (arr[k][j]) {
						cnt++; break;
					}
				}
				for (int k = 0; k < j; k++) {
					if (arr[i][k]) {
						cnt++; break;
					}
				}
				for (int k = j + 1; k < c; k++) {
					if (arr[i][k]) {
						cnt++; break;
					}
				}
			}
		}
	}
	cout << cnt << endl;
}
