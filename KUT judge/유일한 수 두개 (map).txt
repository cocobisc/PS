#include <iostream>
#include <map>
using namespace std;

int main() {
	int t; cin >> t;
	while (t--) {
		map<int, int> m;
		int n; cin >> n;
		for (int i = 0; i < n; i++) {
			int a; cin >> a;
			if (m.find(a) == m.end()) m.insert({ a, 0 });
			else m.erase(a);
		}
		for (auto it = m.begin(); it != m.end(); it++) {
			cout << it->first << " ";
		}
		cout << endl;
	}
}
