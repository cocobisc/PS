#include <cstdio>
#include <cstring>
long long dp[1000001];
const int MOD = 1e9 + 7;
int main() {
	int T; scanf("%d", &T);
	while (T--) {
		memset(dp, 0, sizeof(dp));
		int n, k; scanf("%d %d", &n, &k);
		long long sum = 1;
		dp[n] = 1;
		for (int i = n + 1; i <= k; i++) {
			dp[i] = (sum - dp[i - n - 1]) % MOD;
			sum += (dp[i] - dp[i - n - 1]) % MOD;
		}
		printf("%lld\n", dp[k]);
	}
}
