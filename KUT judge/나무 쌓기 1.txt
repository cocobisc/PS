#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int b[51][51];
int main() {
	int x, y, result = 0;
	cin >> x >> y;
	for (int i = 0; i < y; i++) {
		int max = 0;
		for (int j = 0; j < x; j++) {
			cin >> b[i][j];
			if (b[i][j]) result++;
			if (b[i][j] > max) max = b[i][j];
		}
		result += max;
	}
	for (int i = 0; i < x; i++) {
		int max = 0;
		for (int j = 0; j < y; j++) {
			if (b[j][i] > max) max = b[j][i];
		}
		result += max;
	}
	cout << result << endl;
}
