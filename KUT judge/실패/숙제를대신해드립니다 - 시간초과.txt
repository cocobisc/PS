#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <queue>
#include <cstdio>
#include <string>
#include <vector>
#define MAX 501
#define MIN 1000001
#define swap(a,b) {char temp = a; a = b; b = temp;}
typedef unsigned long long ull;
using namespace std;
int n, k;
int bot[11] = {};
queue<pair<int, int>> q;
vector<pair<int, int>> v;
int main() {
	cin >> n >> k;
	priority_queue<int, vector<int>, greater<int>> pq;
	int t = n;
	int clock = 0;
	for (int a = 0; a < k; a++) {
		int i, j;
		cin >> i >> j;
		q.push(pair<int, int>(i, j));
		v.push_back(pair<int, int>(i, j));
	}
	int max = 0;
	for (int i = 0; i < k; i++) {
		if (bot[i % n] <= v[i].first) bot[i%n] = v[i].first + v[i].second;
		else bot[i % n] += v[i].second;
		if (max < bot[i % n]) max = bot[i % n];
	}
	int te = max;
	max = 0;
	while (!q.empty()) {
		while (!pq.empty() && pq.top() == clock) {
			n++;
			pq.pop();
		}
		while (!q.empty() && n && q.front().first <= clock) {
			int temp = q.front().second + clock;
			pq.push(temp);
			if (max < temp) max = temp;
			q.pop();
			n--;
		}
		clock++;
	}
	cout << (te - max) * 100 << endl;

	return 0;
}
