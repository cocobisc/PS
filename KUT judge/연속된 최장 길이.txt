#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
using namespace std;
int main() {
	int t; scanf("%d",&t);
	while (t--) {
		long arr[10];
		int c = 1, n, max = 1;
		cin >> n;
		for (int i = 0; i < n; i++) cin >> arr[i];
		for (int i = 1; i < n; i++) {
			if (arr[i - 1] + 1 == arr[i]) c++;
			else c = 1;
			if (max < c) max = c;
		}
		if (n == 0) max = 0;
		cout << max << '\n';
	}
}
