#include <iostream>
#include <cmath>
int main() {
	int T; scanf("%d", &T);
	while (T--) {
		int arr[100], ma = 0, mi;
		int b, k; scanf("%d %d", &b, &k);
		for (int i = 0; i < b; i++) {
			scanf("%d", &arr[i]);
			if (ma < arr[i]) ma = arr[i];
		}
		ma *= 2, mi = 0;
		while (mi + 1 < ma) {
			int mid = (mi + ma) / 2;
			int sum = 0;
			for (int i = 0; i < b; i++) sum += ceil((double)arr[i] / mid);
			if (sum <= k) ma = mid;
			else mi = mid;
		}
		printf("%d\n", ma);
	}
