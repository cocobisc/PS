#include<iostream>
#include<map>
#include<string>
using namespace std;
int T,f;
int main(){
	map<char,char>c;
	c[')']='(',c['}']='{',c[']']='[';
	cin>>T;
	while(T--){
		f=0;
		string s,v;cin>>s;
		for(int i=0;i<s.size();i++){
			if(c.find(s[i])==c.end())v.push_back(s[i]);
			else{
				if(v.empty()||v.back()!=c[s[i]]){f=1;break;}
				v.pop_back();
			}
		}if(v.size())f=1;puts((f?"no":"yes"));
	}
}
