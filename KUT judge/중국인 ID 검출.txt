#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <string>
using namespace std;
string check[6] = { "000001", "000010", "000100", "001000", "010000", "100000" };
int main() {
	int T; cin >> T;
	while (T--) {
		string s; cin >> s;
		string local = s.substr(0, 6), date = s.substr(6, 8), order_s = s.substr(14, 3);
		string year_s = date.substr(0, 4), month_s = date.substr(4, 2), day_s = date.substr(6, 2);
		int year = atoi(year_s.c_str()), month = atoi(month_s.c_str()), day = atoi(day_s.c_str());
		int order = atoi(order_s.c_str());
		bool valid = true, sex;
		int i = 0;
		for (; i < 6; i++) {
			if (local == check[i]) break;
		}
		if (i == 6) valid = false;
		if (year < 1900 || year > 2014) valid = false;
		if (month < 1 || month > 12) valid = false;
		if (day < 1 || day > 31) valid = false;
		else {
			if ((month == 1 || month == 3 || month == 5 || month == 7 || month == 8 || month == 10 || month == 12) && day == 30) 
					valid = false;
			if ((month == 4 || month == 6 || month == 9 || month == 11) && day == 31)
					valid = false;
			if (month == 2) {
				if (day == 29)
					if (!((year % 4 == 0 && year % 100 != 0) || year % 400 == 0)) valid = false;
				else if (day > 29) valid = false;
			}

		}
		if (!order) valid = false;
		else {
			if (order % 2) sex = false;
			else sex = true;
		}
		int TP[18]; TP[0] = 1;
		for (int i = 1; i <= 17; i++) TP[i] = (TP[i - 1] * 2);

		int x, b = 0;
		if (s[17] == 'X')
			x = 10;
		else
			x = s[17] - '0';
		for (int i = 0; i < 17; i++) {
			b += (s[i] - '0') * TP[17 - i];
		}
		b %= 11;
		if (x != b) valid = false;
		if (!valid) cout << "Invalid" << '\n';
		else {
			if (sex) cout << "Female" << '\n';
			else cout << "Male" << '\n';
		}
	}
}
