#include <iostream>
#include <algorithm>
#include <string>
#define MAX 10000
using namespace std;
string s;
int dp[5001][5001];

int main() {
	cin >> s;
	string rs;
	int siz = s.size();
	for (int i = siz - 1; i >= 0; i--) rs.push_back(s[i]);
	for (int i = 1; i <= siz; i++) {
		for (int j = 1; j <= siz; j++) {
			if (s[i - 1] == rs[j - 1]) dp[i][j] = dp[i - 1][j - 1] + 1;
			else dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
		}
	}
	cout << s.size() - dp[siz][siz] << endl;
	
	return 0;
}
