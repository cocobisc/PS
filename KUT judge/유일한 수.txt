#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
#include <vector>
#include <cmath>
using namespace std;

int main() {
	int n;
	cin >> n;
	for (; n--;) {
		vector<int> v;
		int a, b, i;
		bool flag = false;
		cin >> a;
		i = a;
		for (; i--;) {
			cin >> b;
			v.push_back(b);
		}
		sort(v.begin(), v.end(), less<int>());
		for (int i = 0; i < a - 1; i += 2) {
			if (v[i] != v[i + 1]) {
				cout << v[i] << endl;
				flag = true;
				break;
			}
		}
		if (!flag) cout << v[a - 1] << endl;
	}
	return 0;
}
