#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <string>
using namespace std;
string a, b;
int main() {
	cin >> a >> b;
	int maxa, maxb, mina, minb;
	for (int i = 0; i < a.size(); i++) if (a[i] == '6') a[i] = '5';
	mina = atoi(a.c_str());
	for (int i = 0; i < a.size(); i++) if (a[i] == '5') a[i] = '6';
	maxa = atoi(a.c_str());
	for (int i = 0; i < b.size(); i++) if (b[i] == '6') b[i] = '5';
	minb = atoi(b.c_str());
	for (int i = 0; i < b.size(); i++) if (b[i] == '5') b[i] = '6';
	maxb = atoi(b.c_str());
	
	cout << mina + minb << " " << maxa + maxb << endl;
}
