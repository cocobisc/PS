#include <cstdio>
#include <iostream>
using namespace std;

int main() {
	int *arr = new int[1000001];
	for (int i = 2; i <= 1000000; i++) arr[i] = i;
	for (int i = 2; i <= 1000000; i++) {
		if (!arr[i]) continue;
		for (int j = i + i; j <= 1000000; j += i) arr[j] = 0;
	}

	int num, n;
	cin >> num;
	for (int i = 0; i < num; i++) {
		int cnt = 0;
		cin >> n;
		for (int i = 2; i <= n - i; i++) {
			if (arr[i] && arr[n - i]) cnt += 1;
		}
		cout << cnt << endl;
	}
	return 0;
}
