#include <iostream>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

long long fac(long long j, int n) {
	long long ret = j;
	for (long long i = j - 1; i > j - n; i--) ret *= i;
	return ret;
}
int main() {
	int num;
	cin >> num;
	for (int i = 0; i < num; i++) {
		int n;
		long long k;
		cin >> n >> k;
		for (long long j = n; ; j++) {
			if (fac(j, n) == k) {
				cout << j - n;
				break;
			}
		}
		cout << "\n";
	}
	return 0;
}
