#include <iostream>
using namespace std;
int arr[10001], n, i;
int visited[10001];
bool dfs(int start) {
	if (start == 0) return false;
	if (start > n) {
		cout << "Yes" << endl;
		exit(0);
	}
	if (visited[start]) return false;
	visited[start] = true;
	for (int i = arr[start]; i >= 1; i--) {
		start += i;
		dfs(start);
		start -= i;
	}
}
int main() {
	cin >> n;
	for (int i = 1; i <= n; i++) cin >> arr[i];
	dfs(1);
	cout << "No" << endl;
}
