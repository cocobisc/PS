#include <algorithm>
#include <iostream>
#include <string>
using namespace std;
string s[100000];
bool cmp(string a, string b) {
	for (int idx1 = 0, idx2 = 0; idx1 < a.size() && idx2 < b.size(); idx1++, idx2++) {
		if (!isdigit(a[idx1]) && !isdigit(b[idx2]) && a[idx1] == b[idx2]) continue;
		if (!isdigit(a[idx1]) && isdigit(b[idx2])) return false;
		if (isdigit(a[idx1]) && !isdigit(b[idx2])) return true;
		if (!isdigit(a[idx1]) && !isdigit(b[idx2])) return a[idx1] < b[idx2];
		if (isdigit(a[idx1]) && isdigit(b[idx2])) {
			string num1, num2;
			while (idx1 < a.size() && isdigit(a[idx1])) { num1 += a[idx1++]; }
			while (idx2 < b.size() && isdigit(b[idx2])) { num2 += b[idx2++]; }
			long long n1 = atoll(num1.c_str()), n2 = atoll(num2.c_str());
			if (n1 == n2) {
				idx1--; idx2--; continue;
			}
			else return atoll(num1.c_str()) < atoll(num2.c_str());
		}
	}
	return false;
}

int main() {
	int n; cin >> n;
	for (int i = 0; i < n; i++) cin >> s[i];
	sort(s, s + n, cmp);
	for (int i = 0; i < n; i++) cout << s[i] << '\n';
}
