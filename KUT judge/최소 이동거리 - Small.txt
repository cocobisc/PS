#include <iostream>
#include <algorithm>
#include <string>
#include <cmath>

using namespace std;
int n, arr[100];

int main() {
	int min = 1000001;
	cin >> n;
	for (int i = 0; i < n; i++) cin >> arr[i];
	for (int i = 0; i < n; i++) {
		int sum = 0;
		for (int j = 0; j < n; j++) {
			if (i == j) continue;
			sum += abs(arr[i] - arr[j]);
		}
		if (sum < min) min = sum;
	}
	cout << min << endl;
	
	return 0;
}
