#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int b[51][51];
int main() {
	int x, y, result = 0;
	cin >> x >> y;
	for (int i = 0; i < y; i++) {
		for (int j = 0; j < x; j++) {
			cin >> b[i][j];
			if (b[i][j]) {
				result += 4 * b[i][j];
				result += 1;
			}
		}
	}
	for (int i = 0; i < y; i++) {
		for (int j = 0; j < x - 1; j++) {
			if(b[i][j] < b[i][j+1]) result -= 2 * b[i][j];
			else result -= 2 * b[i][j + 1];
		}
	}
	for (int j = 0; j < x; j++) {
		for (int i = 0; i < y - 1; i++) {
			if (b[i][j] < b[i + 1][j]) result -= 2 * b[i][j];
			else result -= 2 * b[i + 1][j];
		}
	}
	cout << result << endl;
}
