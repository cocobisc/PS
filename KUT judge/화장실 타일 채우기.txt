#define _CRT_SECURE_NO_WARNINGS
// #include <bits/stdc++.h>
#include <iostream>
using namespace std;
int n;
int dp[101];

int main() {
	dp[0] = 1;
	dp[1] = 1;
	for (int i = 2; i <= 100; i++) {
		dp[i] = dp[i - 1] + dp[i - 2];
		dp[i] %= 1000000007;
	}

	cin >> n;
	for (int i = 0; i < n; i++) {
		int a;
		cin >> a;
		cout << dp[a] << endl;
	}
	
	return 0;
}
